### :rocket: 이진탐색

  const binarySearch = (array, target) => {
    let [start, end] = [0, array.length - 1];

    while (start <= end) {
      let mid = Math.floor((start + end) / 2);

      if (array[mid] === target) {
        return mid;
      }

      if (target <= array[mid]) {
        end = mid - 1;
      } else {
        start = mid + 1;
      }
    }
    return -1;
  };
